// app/layout.tsx - DEBUG VERSION
'use client';

import './globals.css';
import { useState, useEffect } from 'react';
import { usePathname } from 'next/navigation';
import SplashScreen from '../components/SplashScreen';
import DesktopLayout from '../components/DesktopLayout';
import MobileLayout from '../components/MobileLayout';
import MobileMasterView from '../components/MobileMasterView';

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  const [isMobile, setIsMobile] = useState(false);
  const [showSplash, setShowSplash] = useState(true);
  const pathname = usePathname();

  console.log('Layout rendering - showSplash:', showSplash, 'pathname:', pathname);

  useEffect(() => {
    console.log('Layout useEffect running - this should happen on every refresh');
    
    const checkMobile = () => {
      setIsMobile(window.innerWidth < 768);
    };
    
    checkMobile();
    window.addEventListener('resize', checkMobile);

    // Force show splash on every load
    setShowSplash(true);
    console.log('Splash screen should be visible now');

    const timer = setTimeout(() => {
      console.log('Hiding splash screen after timeout');
      setShowSplash(false);
    }, 6800);
    
    return () => {
      console.log('Layout cleanup');
      clearTimeout(timer);
      window.removeEventListener('resize', checkMobile);
    };
  }, [pathname]); // Added pathname to dependencies

  const renderContent = () => {
    if (isMobile) {
      if (pathname === '/mobile-master') {
        return <MobileMasterView />;
      } else {
        return <MobileLayout />;
      }
    } else {
      return <DesktopLayout />;
    }
  };

  return (
    <html>
      <body className="h-screen w-screen overflow-hidden">
        {showSplash ? (
          <>
            <div style={{position: 'absolute', top: 10, left: 10, background: 'red', color: 'white', padding: 10, zIndex: 1000}}>
              DEBUG: Splash Visible
            </div>
            <SplashScreen />
          </>
        ) : (
          <>
            <div style={{position: 'absolute', top: 10, left: 10, background: 'green', color: 'white', padding: 10, zIndex: 1000}}>
              DEBUG: Content Visible
            </div>
            {renderContent()}
          </>
        )}
      </body>
    </html>
  );
}
